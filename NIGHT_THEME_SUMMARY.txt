## How to Use

### In Components
```tsx
import { useTheme } from './context/ThemeContext';

function MyComponent() {
  const { isDark, toggleTheme } = useTheme();
  
  return (
    <div className="night-card">
      <h2 className="text-night-text">Title</h2>
      <p className="text-night-text-secondary">Description</p>
      <button className="night-btn">Action</button>
    </div>
  );
}
```

### Available Classes

BACKGROUNDS:        bg-night-{darker,dark,base,card,hover}
TEXT:              text-night-{text,text-secondary,text-subtle}
ACCENTS:           text-night-{green,accent,red,orange}
COMPONENTS:        .night-card, .night-btn, .night-input
EFFECTS:           .night-glow, .night-text-gradient
SHADOWS:           shadow-night-{sm,md,lg,glow,glow-blue}
BADGES:            .night-badge, .night-badge.discount, .night-badge.hot
ALERTS:            .night-alert.{success,error,warning,info}
